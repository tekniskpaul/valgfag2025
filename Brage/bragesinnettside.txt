<script>
const games = {
    cs2:{name:"CS2", url:"https://store.steampowered.com/app/730/CounterStrike_2/", pass:true, img:"https://cdn.cloudflare.steamstatic.com/steam/apps/730/header.jpg"},
    cod:{name:"Call of Duty", url:"https://www.callofduty.com/", pass:true, img:"https://www.callofduty.com/content/dam/atvi/callofduty/cod-touchui/blog/body/mwiii/MWIII-REVEAL-FULL-TOUT.jpg"},
    tarkov:{name:"Tarkov", url:"https://www.escapefromtarkov.com/", pass:true, img:"https://www.escapefromtarkov.com/uploads/news/escape-from-tarkov.jpg"},
    shredders:{name:"Shredders", url:"https://store.steampowered.com/app/1874170/Shredders/", pass:true, img:"https://cdn.cloudflare.steamstatic.com/steam/apps/1874170/header.jpg"},

    fortnite:{name:"Fortnite", url:"https://www.fortnite.com/", pass:false, img:"https://cdn2.unrealengine.com/fortnite.jpg"},
    minecraft:{name:"Minecraft", url:"https://www.minecraft.net/", pass:false, img:"https://www.minecraft.net/content/dam/minecraft.jpg"},
    gta:{name:"GTA V", url:"https://www.rockstargames.com/gta-v", pass:false, img:"https://www.rockstargames.com/V/img/global/share.jpg"},
    valorant:{name:"Valorant", url:"https://playvalorant.com/", pass:false, img:"https://playvalorant.com/assets/hero.jpg"},
    apex:{name:"Apex", url:"https://www.ea.com/games/apex-legends", pass:false, img:"https://media.contentapi.ea.com/content/dam/apex.jpg"}
};

let current=null;

// BUILD GRID
const grid=document.getElementById("grid");
Object.keys(games).forEach(key=>{
    let g=games[key];

    let div=document.createElement("div");
    div.className="game";
    div.innerHTML=`<img src="${g.img}"><p>${g.name}</p>`;
    div.onclick=()=>selectGame(key);

    grid.appendChild(div);
});

// SELECT
function selectGame(key){
    current=key;
    document.getElementById("selected").innerText=games[key].name;
    document.getElementById("status").innerText="READY: "+key;
}

// SEARCH
search.addEventListener("input",()=>{
    let val=search.value.toLowerCase();
    let s=document.getElementById("suggestions");
    s.innerHTML="";

    Object.keys(games).filter(g=>g.startsWith(val)).forEach(g=>{
        let d=document.createElement("div");
        d.innerText=g;
        d.onclick=()=>{selectGame(g); s.innerHTML="";}
        s.appendChild(d);
    });
});

// ENTER
document.addEventListener("keydown",e=>{
    if(e.key==="Enter") launch();
});

// LAUNCH
function launch(){
    if(!current){
        status.innerText="❌ no game selected";
        return;
    }

    let p=password.value.toLowerCase();

    if(games[current].pass && p!==current){
        status.innerText="❌ wrong password";
        return;
    }

    document.getElementById("loading").style.display="block";

    let w=0;
    let int=setInterval(()=>{
        w++;
        progress.style.width=w+"%";
        percent.innerText=w+"%";

        if(w>=100){
            clearInterval(int);
            window.location.href=games[current].url;
        }
    },20);
}
